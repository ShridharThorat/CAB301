\section*{First-Come First-Served Algorithm}
\begin{algorithm}
    \caption*{\textbf{ALGORITHM~~}  FirstComeFirstServed{()}}
    \textbf{Output:} An array $IJob[~]$ which contains all the jobs in a scheduler and they are sorted in non-descending order of the $timeReceived$ of the job

    \begin{algorithmic}[1]
        \State{$jobs \gets$ convert $Jobs$ to array} 

        \State{$collection \gets$ new JobCollection with same capacity as $jobs$}
        \State{\texttt{// Consider that $n$ represents the number of jobs in a $collection$}}

        \ForAll{$job$ in $jobs$}
            \State{add $job$ to $collection$}
        \EndFor{} 
        \For{$i \gets 0$ to $n-2$}
            \State{$\min \gets i$}
            \For{$j \gets i+1$ to $n-1$}
                \If{$pJobs[j].timeReceived < pJobs[\min].timeReceived$}
                    \State{$\min \gets j$}
                \EndIf{}
            \EndFor{}
            \State{Swap $jobs[i]$ and $jobs[\min]$}
        \EndFor{}
        \State{\Return{$pJobs$}}

    \end{algorithmic}
\end{algorithm}
